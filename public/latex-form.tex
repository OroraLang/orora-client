\newenvironment{for}[2]
{
  \par\noindent\textbf{for} #1 $\in$ #2  \textbf{do}
  \begin{quote}\ttfamily
}
{
  \end{quote}
  \par\noindent\textbf{end for}
}

\newenvironment{while}[1]
{
  \par\noindent\textbf{while} #1 \textbf{do}
  \begin{quote}\ttfamily
}
{
  \end{quote}
  \par\noindent\textbf{end while}
}
